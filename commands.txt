# From /Users/dmccanns/Desktop/Tinfoil/tinfoil-bucket

# 1. Resolve deps + generate go.sum
go mod tidy

# 2. Verify it compiles
go build .

# 3. Run locally (sidecar must be reachable at http://localhost:9000).
#    USAGE_REPORTER_SECRET unset = reporter no-ops, which is what you want for local.
LISTEN_ADDR=":8080" \
CONTROL_PLANE_URL="https://api.tinfoil.sh" \
USAGE_REPORTER_ID="tinfoil-bucket" \
go run .

# 4. Smoke-test (in a second shell). Replace API_KEY and KEY.
API_KEY="sk-tinfoil-..."
KEY=$(openssl rand -base64 32)
curl -v -X PUT \
  -H "Authorization: Bearer $API_KEY" \
  -H "X-Tinfoil-Encryption-Key: $KEY" \
  --data-binary @./README.md \
  http://localhost:8080/any-bucket/hello.txt

# 5. Docker build (mirrors how the enclave will build it)
docker build -t tinfoil-bucket .
